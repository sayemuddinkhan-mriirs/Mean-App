import { Injectable } from '@angular/core';
import { HttpClient , HttpHeaders} from '@angular/common/http';
//import { Contact } from './contact'

@Injectable({
  providedIn: 'root'
})
export class ContactService {

  constructor(private http : HttpClient) { }

  getContacts(){
    return this.http.get('http://localhost:3000/api/contacts');
  }

  createContacts(contact: any){
    var headers = new HttpHeaders();
    headers.append('Content-Type','application/json');

    return this.http.post('http://localhost:3000/api/contact',contact,{headers:headers});

  }

  updateContacts(contact: any){
    return this.http.put('http://localhost:3000/api/contact/' +contact._id,contact);
  }

  deleteContacts(id: any){
    return this.http.delete('http://localhost:3000/api/contact/'+id);

  }
}



export class Contact {
    _id?: string ; 
    first_name: string;
    last_name: string;
    phone: string ;
}
